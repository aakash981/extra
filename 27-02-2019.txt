    { 
        DateTime i=dt.AddDays(1);
        DateTime j = dt.AddDays(-1);
        while (i.DayOfWeek != DayOfWeek.Friday) {
         i=i.AddDays(1);
                           }
        while (j.DayOfWeek != DayOfWeek.Friday)
        {
            j = j.AddDays(-1);       
        }
       
        Console.WriteLine("Next Friday : " + i.ToString("yyyy-MM-dd"));
